为了更好的提高系统性能，多个线程可能会并发调用日志类的push接口，向单例logger类中的日志队列中添加元素，进行序列化后调用write系统调用写入磁盘中。

一、传统阻塞队列实现

#ifndef LOCKED_QUEUE_H
#define LOCKED_QUEUE_H

#include <queue>
#include <mutex>
#include <condition_variable>
#include <chrono>
#include <ctime>
#include <fstream>

template <typename T>
class LockedQueue {
public:
    void push(const T& value) {
        std::unique_lock<std::mutex> lock(mutex_);
        queue_.push(value);
        cv_.notify_one(); // 通知等待的线程有新元素
    }

    bool try_pop(T& value) {
        std::unique_lock<std::mutex> lock(mutex_);
        cv_.wait(lock, [this]() { return !queue_.empty(); }); // 等待直到队列非空

        if (queue_.empty()) {
            return false; // 虽然通知了，但队列为空，返回false
        }

        value = queue_.front();
        queue_.pop();
        writeLogToDisk(value); // 写入日志到磁盘
        return true;
    }

    bool empty() const {
        std::unique_lock<std::mutex> lock(mutex_);
        return queue_.empty();
    }

    int size() const {
        std::unique_lock<std::mutex> lock(mutex_);
        return queue_.size();
    }

private:
    std::queue<T> queue_;
    mutable std::mutex mutex_;
    std::condition_variable cv_;

    void writeLogToDisk(const T& value) {
        std::ofstream logFile("consumed_log.txt", std::ios::app);
        if (logFile.is_open()) {
            auto now = std::chrono::system_clock::now();
            std::time_t time = std::chrono::system_clock::to_time_t(now);
            logFile << "Consumed: " << value << " at " << std::ctime(&time);
            logFile.close();
        }
    }
};

#endif // LOCKED_QUEUE_H

上述是经典的使用锁和条件变量实现多线程并发的实现，在实现中会开启一个无限循环的工作位列一直尝试执行try_pop函数写入。

注意在执行队列检查的时候需要进行两次检查，避免线程的虚假唤醒的问题，这里的原理是锁唤醒可能是虚假唤醒问题，这种检查模式被称为虚假唤醒（Spurious Wakeup）。

cv_.wait(lock, [this]() { return !queue_.empty(); }); // 等待直到队列非空
if (queue_.empty()) {

}

虚假唤醒（Spurious Wakeup）
考虑到多生产者和多消费者的问题，条件变量condition_variable通常需要和互斥锁mutix共同使用，在使用wait函数进行检查的时候，线程会先获取mutex互斥锁，并进行资源检查queue_.empty()，如果条件不满足则会释放锁，线程阻塞释放CPU资源，等待对应的cv条件变量进行唤醒。

cv_.notify_one();

上述提供的notify_one()尝试唤醒等待这个条件变量的陷入阻塞的线程。

C++中的虚假唤醒可能由许多因素引起，具体取决于条件变量的实现细节。一个常见的罪魁祸首是某些编译器可能会重排序原子指令，使得存储操作在唤醒操作之后完成。这可能导致通知操作先发生，然后存储释放操作在其后发生，从而导致过早唤醒。以下是可能发生的场景：

场景描述：
考虑一个生产者-消费者队列，有三个线程：

Thread 1（消费者线程）：已经从队列中取出一个元素并释放了互斥锁，现在正在处理这个元素。
Thread 2（消费者线程）：尝试从队列中取出元素，但发现队列为空，调用 pthread_cond_wait 阻塞等待通知。
Thread 3（生产者线程）：获取互斥锁，向队列中插入一个新元素，通知条件变量，然后释放互斥锁。
具体过程：
Thread 1：
已经从队列中取出一个元素并释放了互斥锁。
现在正在处理这个元素。

Thread 2：
尝试从队列中取出元素，但发现队列为空。
调用 pthread_cond_wait 阻塞等待通知。

Thread 3：
获取互斥锁，向队列中插入一个新元素。
通知条件变量（调用 pthread_cond_signal 或 pthread_cond_broadcast）。
释放互斥锁。

Thread 2 被唤醒：
响应 Thread 3 的通知，Thread 2 被调度运行。
但在 Thread 2 获取队列锁之前，Thread 1 完成了当前任务并返回队列请求更多工作。

Thread 1 继续工作：
获取队列锁，检查条件变量，发现队列中有新元素。
继续从队列中取出 Thread 3 插入的元素，释放互斥锁，并处理这个元素。

Thread 2 获取锁：
现在 Thread 2 获取到队列锁。
检查条件变量时，发现队列为空（Thread 1 已经取走了元素）。
这时，Thread 2 的唤醒看起来是虚假的。

上述对条件变量pthread_cond的设置和signal唤醒并不是原子指令的，可能存在指令重排的情况导致虚假唤醒这也是C++11提供的condition_variale需要搭配lock(loack_guard或者unique_lock)的原因。后续会专门写一篇笔记探究上述的虚假唤醒的问题。

eg.可以使用双重队列的做法，设定日志队列的阈值，超过阈值时执行write写入，并更换logs日志队列，实现入盘的优化。

二、无锁日志队列的实现
无锁队列的实现依赖于具体的无锁数据结构，通过操作系统提供的原子指令和内存模型提供多线程之间的同步，在无锁日志队列中根据具体的使用场景可以进行以下划分：
1、多生产者单消费者模式
这种模式与单例模式可以很好的融合多个生产者进行push单个消费者消费日志落盘，以下是一个无锁队列的实现：

#include <atomic>
#include <memory>
#include <iostream>

template<typename T>
class lock_free_queue
{
private:
    struct node
    {
        std::shared_ptr<T> data;
        std::atomic<node*> next;

        node(T data_) : data(std::make_shared<T>(data_)), next(nullptr) {}
        node() : data(nullptr), next(nullptr) {}
    };

    alignas(64) std::atomic<node*> head; // 对齐以避免假共享
    alignas(64) std::atomic<node*> tail; // 对齐以避免假共享
    alignas(64) std::atomic<std::size_t> count; // 记录节点个数

    node* pop_head()
    {
        node* old_head = head.load(std::memory_order_relaxed);
        while (old_head && !head.compare_exchange_weak(old_head, old_head->next.load(), std::memory_order_release, std::memory_order_relaxed));
        return old_head;
    }

public:
    lock_free_queue() : head(new node), tail(head.load()), count(0) {}
    ~lock_free_queue()
    {
        while (node* old_head = head.load())
        {
            head.store(old_head->next, std::memory_order_relaxed);
            delete old_head;
        }
    }

    void push(T new_value)
    {
        std::shared_ptr<T> new_data(std::make_shared<T>(new_value));
        node* p = new node;
        node* old_tail = tail.load(std::memory_order_relaxed);

        node* new_next = nullptr; // 使用局部变量来存储 nullptr
        while (!old_tail->next.compare_exchange_weak(new_next, p, std::memory_order_release, std::memory_order_relaxed)) {
            new_next = nullptr; // 重置 new_next 以避免循环中使用旧值
            // 重新加载 old_tail 以避免 ABA 问题
            old_tail = tail.load(std::memory_order_relaxed);
        }

        // 更新 tail 指针
        tail.compare_exchange_strong(old_tail, p, std::memory_order_release, std::memory_order_relaxed);

        // 将数据赋值给 old_tail 节点
        old_tail->data = new_data;

        // 增加节点计数
        count.fetch_add(1, std::memory_order_relaxed);
    }

    std::shared_ptr<T> pop()
    {
        node* old_head = pop_head();
        if (!old_head)
        {
            return std::shared_ptr<T>();
        }
        std::shared_ptr<T> res = old_head->data;
        delete old_head;

        // 减少节点计数
        count.fetch_sub(1, std::memory_order_relaxed);

        return res;
    }

    // 返回队列大小
    std::size_t size() const
    {
        return count.load(std::memory_order_relaxed);
    }
};

因为只有单个线程调用pop操作，不存在并发场景，因此获得节点后可以直接进行delete操作删除节点,实现上述的关键在于push操作，关键点在于

while (!old_tail->next.compare_exchange_weak(new_next, p, std::memory_order_release, std::memory_order_relaxed))；

保证只有一个线程获得队列的尾部tail，同时将tail->next设置为新创建的节点，注意此时队列的tail指针未发生改变，注意这里需要进行循环load操作更新队列的尾指针tail，防止以为其他的线程进行更新。

后续通过compare_exchange_strong函数实现原子的更新队列的尾部，后续将创建好的数据赋值为先前获得old_tail节点(确保了尾部节点的数据指针始终为nullptr)

tail.compare_exchange_strong(old_tail, p, std::memory_order_release, std::memory_order_relaxed);

通过上述无锁编程实现了多生成者和单消费者的无锁队列模型。后续会实现多生产者多消费者的无锁队列模型，涉及到节点的生命周期、内存释放问题，更具有挑战性。
