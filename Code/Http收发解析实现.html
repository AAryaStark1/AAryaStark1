本章笔记是对http请求的接收和解析过程的一个笔记，分为使用Epoll多路复用和io_uring进行http请求解析实现。

一、io_uring异步实现

auto Scheduler::receive(const Client &client, const std::source_location sourceLocation) -> Task {
    std::vector<std::byte> receiveBuffer;

    while (true) {
        if (const auto [result, flags]{co_await client.receive(this->ringBuffer.getId())};
            result > 0 && (flags & IORING_CQE_F_MORE) != 0) {
            const auto index{static_cast<unsigned short>(flags >> IORING_CQE_BUFFER_SHIFT)};
            const std::span buffer{this->bufferGroup.getBuffer(index)};
            const std::span receivedData{buffer.first(result)};
            this->ringBuffer.addBuffer(buffer, index);

            receiveBuffer.insert(receiveBuffer.cend(), receivedData.cbegin(), receivedData.cend());

            if ((flags & IORING_CQE_F_SOCK_NONEMPTY) == 0) {
                this->timer.update(client.getFileDescriptor(), client.getSeconds());

                std::vector response{this->httpParse.parse(
                    std::string_view{reinterpret_cast<const char *>(receiveBuffer.data()), receiveBuffer.size()})};
                receiveBuffer.clear();

                this->submit(std::make_shared<Task>(this->send(client, std::move(response))));
            }
        } else {
            this->logger->push(Log{
                Log::Level::warn,
                result == 0 ? "connection closed" :
                              std::error_code{std::abs(result), std::generic_category()}
                              .message(),
                sourceLocation
            });

            this->timer.remove(client.getFileDescriptor());
            this->submit(std::make_shared<Task>(this->close(client.getFileDescriptor())));

            break;
        }
    }

    this->eraseCurrentTask();
}

基于io_uring和协程的http的异步编程可以实现以同步的逻辑进行代码实现。通过对对应缓存区的接收数据进行拷贝，复制到receiveBuffer接收缓冲区中，通过IORING_CQE_F_SOCK_NONEMPTY标志位判断是否已经接收完成，更新客户端的定时器计时，调用httpParse的主要逻辑对缓冲区中的数据进行解析，并将结果写回到response的用户缓冲区中，并通过创建send异步协程进行客户端的发送处理。httpParse的执行逻辑，将在后续的章节中进行讲解。

二、基于opoll的实现驱动

void WebServer::Start() {
    int timeMS = -1;  /* epoll wait timeout == -1 无事件将阻塞 */
    if(!isClose_) { LOG_INFO("========== Server start =========="); }
    while(!isClose_) {
        if(timeoutMS_ > 0) {
            timeMS = timer_->GetNextTick();
        }
        int eventCnt = epoller_->Wait(timeMS);
        for(int i = 0; i < eventCnt; i++) {
            int fd = epoller_->GetEventFd(i);
            uint32_t events = epoller_->GetEvents(i);
            if(fd == listenFd_) {
                DealListen_();
            }
            else if(events & (EPOLLRDHUP | EPOLLHUP | EPOLLERR)) {
                assert(users_.count(fd) > 0);
                CloseConn_(&users_[fd]);
            }
            else if(events & EPOLLIN) {
                assert(users_.count(fd) > 0);
                DealRead_(&users_[fd]);
            }
            else if(events & EPOLLOUT) {
                assert(users_.count(fd) > 0);
                DealWrite_(&users_[fd]);
            } else {
                LOG_ERROR("Unexpected event");
            }
        }
    }
}

通过timeMS = timer_->GetNextTick()获得最短过期的客户端链接，进行过期客户端的处理，通过调用Wait函数实现对epoll_wait函数调用的封装，获得就绪队列以及长度。查看对应的事件类型进行相应的处理，包括对错误客户端的关闭，接收事件，读写事件，其中后续主要查看读事件。

void WebServer::DealRead_(HttpConn* client) {
    assert(client);
    ExtentTime_(client);
    threadpool_->AddTask(std::bind(&WebServer::OnRead_, this, client));
}

void WebServer::OnRead_(HttpConn* client) {
    assert(client);
    int ret = -1;
    int readErrno = 0;
    ret = client->read(&readErrno);
    if(ret <= 0 && readErrno != EAGAIN) {
        CloseConn_(client);
        return;
    }
    OnProcess(client);
}
void WebServer::OnProcess(HttpConn* client) {
    if(client->process()) {
        epoller_->ModFd(client->GetFd(), connEvent_ | EPOLLOUT);
    } else {
        epoller_->ModFd(client->GetFd(), connEvent_ | EPOLLIN);
    }
}

读事件中，将一个仿函数打包发送到线程池中，线程池是典型的通过锁和条件变量进行数据同步的阻塞队列，逻辑执行后调用OnRead_函数，进行对应客户端的读取，调用OnProcess对客户端的缓冲区数据进行解析处理。client->process()的代码中会执行具体的解析实现。

上述实现的差异，很大一部分在对解析部分时机的控制，在io_uring实现的http请求的解析中，可以通过标志位判断是否已经读取完成，而对于Epoll实现事件触发，当存在读时间后都会触发上述的一套流程，因此需要将先前解析到的内容保留到客户端的数据缓冲区中。IO_uring则是在解析之前进行数据保留，接收完成后再进行一次性的读取。

三、基于IO_Uring的具体http解析实现

auto HttpParse::parse(const std::string_view request, const std::source_location sourceLocation)
    -> std::vector<std::byte> {
    try {
        this->httpRequest = HttpRequest{request};
        this->parseVersion();
    } catch (Exception &exception) {
        this->handleException();
        this->logger->push(std::move(exception.getLog()));
    } catch (const std::exception &exception) {
        this->handleException();
        this->logger->push(Log{Log::Level::warn, exception.what(), sourceLocation});
    }

    this->httpResponse.addHeader("Content-Length: " + std::to_string(this->body.size()));

    if (!this->isWriteBody) this->body.clear();
    this->httpResponse.setBody(this->body);

    std::vector response{this->httpResponse.toByte()};
    this->clear();

    return response;
}

首先将接收的字符string_view类型的字符串转换为请求类型request：

HttpRequest::HttpRequest(std::string_view request) {
    bool isParsedLine{}, isParsedBody{};

    for (unsigned long lineBreak{request.find("\r\n")}; lineBreak != std::string_view::npos && !isParsedBody;
         lineBreak = request.find("\r\n")) {
        if (!isParsedLine) {
            isParsedLine = true;
            this->parseLine(request.substr(0, lineBreak));

            request.remove_prefix(lineBreak + 2);
        } else [[unlikely]] if (lineBreak == 0) {
            isParsedBody = true;
            this->body = request.substr(2);
        } else {
            this->parseHeader(request.substr(0, lineBreak));

            request.remove_prefix(lineBreak + 2);
        }
    }
}

1、通过"\r\n"标志将request请求报文分割为请求行，请求头和请求体这三个部分,可以一次性的解析完成，this->httpRequest = HttpRequest{request}。

2、解析版本信息，parseMethod()解析方法，提供了login方法和注册方法

auto HttpParse::parseVersion() -> void {
    if (const std::string_view version{this->httpRequest.getVersion()}; version == "HTTP/1.1") {
        this->httpResponse.setVersion(version);

        this->parseMethod();
    } else {
        this->httpResponse.setVersion("HTTP/1.1");
        this->httpResponse.setStatusCode("505 HTTP Version Not Supported");
    }
}

auto HttpParse::parseMethod() -> void {
    if (const std::string_view method{this->httpRequest.getMethod()}; method == "GET" || method == "HEAD") {
        if (method == "HEAD") this->isWriteBody = false;

        this->parsePath();
    } else if (method == "POST") {
        this->httpResponse.setStatusCode("200 OK");
        this->httpResponse.addHeader("Content-Type: application/json; charset=utf-8");

        const JsonObject requestBody{this->httpRequest.getBody()};
        const std::string_view password{requestBody["password"]};

        JsonObject jsonBody;
        if (std::string_view{requestBody["method"]} == "login") {
            jsonBody.insert("success",
                            JsonValue{!this->database
                                           .query(std::format("SELECT * FROM users WHERE id = {} AND password = '{}';", std::string_view{requestBody["id"]}, password))
                                           .empty()});
        } else {
            this->database.query(std::format("INSERT INTO users (password) VALUES ('{}');", password));

            jsonBody.insert("id", JsonValue{std::move(this->database.query("SELECT LAST_INSERT_ID();")[0][0])});
        }

        const auto stringBody{jsonBody.toString()};
        const auto bytes{std::as_bytes(std::span{stringBody})};
        this->body = std::vector<std::byte>{bytes.cbegin(), bytes.cend()};
    } else this->httpResponse.setStatusCode("405 Method Not Allowed");
}

auto HttpParse::parseResource(const std::string &resourcePath) -> void {
    static constexpr auto maxSize{static_cast<unsigned int>(std::pow(2, 20))};
    const auto resourceSize{static_cast<long>(std::filesystem::file_size(resourcePath))};
    std::pair<long, long> range;

    if (this->httpRequest.containsHeader("Range")) {
        const auto rangeHeader{this->httpRequest.getHeaderValue("Range").substr(6)};
        const unsigned long splitPoint{rangeHeader.find('-')};

        const std::string stringStart{rangeHeader.cbegin(), rangeHeader.cbegin() + splitPoint};
        range.first = std::stol(stringStart);

        std::string stringEnd{rangeHeader.cbegin() + splitPoint + 1, rangeHeader.cend()};
        if (stringEnd.empty()) {
            range.second = range.first + maxSize - 1;
            if (range.second > resourceSize) range.second = resourceSize - 1;

            stringEnd = std::to_string(range.second);
        } else range.second = std::stol(stringEnd);

        if (range.first > range.second || range.second >= resourceSize) {
            this->httpResponse.setStatusCode("416 Range Not Satisfiable");
            this->httpResponse.clearHeaders();

            return;
        }

        this->httpResponse.setStatusCode("206 Partial Content");
        this->httpResponse.addHeader("Content-Range: bytes " + stringStart + '-' + stringEnd + '/' +
                                     std::to_string(resourceSize));
    } else if (resourceSize > maxSize) {
        this->httpResponse.setStatusCode("206 Partial Content");
        this->httpResponse.addHeader("Content-Range: bytes 0-" + std::to_string(maxSize - 1) + '/' +
                                     std::to_string(resourceSize));

        range = std::pair{0, maxSize - 1};
    } else [[likely]] {
        this->httpResponse.setStatusCode("200 OK");

        range = std::pair{0, resourceSize - 1};
    }

    this->readResource(resourcePath, range);
}

由上述实现全部的http请求的解析。

四、基于Epoll的具体http解析实现

void WebServer::OnProcess(HttpConn* client) {
    if(client->process()) {
        epoller_->ModFd(client->GetFd(), connEvent_ | EPOLLOUT);
    } else {
        epoller_->ModFd(client->GetFd(), connEvent_ | EPOLLIN);
    }
}

bool HttpConn::process() {
    request_.Init();
    if(readBuff_.ReadableBytes() <= 0) {
        return false;
    }
    else if(request_.parse(readBuff_)) {
        LOG_DEBUG("%s", request_.path().c_str());
        response_.Init(srcDir, request_.path(), request_.IsKeepAlive(), 200);
    } else {
        response_.Init(srcDir, request_.path(), false, 400);
    }

    response_.MakeResponse(writeBuff_);
    iov_[0].iov_base = const_cast<char*>(writeBuff_.Peek());
    iov_[0].iov_len = writeBuff_.ReadableBytes();
    iovCnt_ = 1;

    /* 文件 */
    if(response_.FileLen() > 0  && response_.File()) {
        iov_[1].iov_base = response_.File();
        iov_[1].iov_len = response_.FileLen();
        iovCnt_ = 2;
    }
    LOG_DEBUG("filesize:%d, %d  to %d", response_.FileLen() , iovCnt_, ToWriteBytes());
    return true;
}

bool HttpRequest::parse(Buffer& buff) {
    const char CRLF[] = "\r\n";
    if(buff.ReadableBytes() <= 0) {
        return false;
    }
    while(buff.ReadableBytes() && state_ != FINISH) {
        const char* lineEnd = search(buff.Peek(), buff.BeginWriteConst(), CRLF, CRLF + 2);
        std::string line(buff.Peek(), lineEnd);
        switch(state_)
        {
        case REQUEST_LINE:
            if(!ParseRequestLine_(line)) {
                return false;
            }
            ParsePath_();
            break;    
        case HEADERS:
            ParseHeader_(line);
            if(buff.ReadableBytes() <= 2) {
                state_ = FINISH;
            }
            break;
        case BODY:
            ParseBody_(line);
            break;
        default:
            break;
        }
        if(lineEnd == buff.BeginWrite()) { break; }
        buff.RetrieveUntil(lineEnd + 2);
    }
    LOG_DEBUG("[%s], [%s], [%s]", method_.c_str(), path_.c_str(), version_.c_str());
    return true;
}

上述实现使用了有限状态机的原理实现,根据实现的状态对http请求做出解析，这里的解析是一次性的，需要对当前读缓冲区的数据进行全部读取，根据设置的标志为进行不同的行为，如果是边缘触发的端口，则不会再次触发。而如果是水平触发的版本，只要缓冲区中有数据则可以进行事件触发。这里需要对当前缓冲区的所有数据流进行读取，如果数据是不完整的，则会发出400的错误码。所以上述的版本需要发送方一次性进行发送，进行完整的解析。

question:如果用户一次性传入的请求太大，或者是不完整的请求信息，需要多次发送才能组成完整的http请求报文，这里如何解决？
对于IO_URING，通过flags & IORING_CQE_F_MORE标志位判断是否对该socket的信息接收完全，接收完全之后则会进行完整的解析。对于epoll事件触发模型的socket端口，对于上述不完整的io端口则没有进行多个数据报文结合的设置，需要不断的解析数据报文判断是否已经完整了，HTTP 请求通常以空行（即两个换行符 \r\n\r\n）结束，如果请求有主体（例如 POST 请求），根据请求头中的 Content-Length字段来确定主体的长度。
