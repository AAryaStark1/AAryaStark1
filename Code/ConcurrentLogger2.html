为了更好的提高系统性能，多个线程可能会并发调用日志类的push接口，向单例logger类中的日志队列中添加元素，进行序列化后调用write系统调用写入磁盘中。考虑更一般的场景，存在多个消费者执行pop操作尝试从队列中获取元素，但是同时只有一个线程会获得，涉及到更复杂的同步问题。

三、无锁日志队列的实现2
这里实现无锁队列的多生产者多消费者模型MPMC，参考C++并发编程实战第二版实现，关键点在于对节点的生命周期进行管理，以下的代码使用了为节点添加引用计数进行管理。

#include <atomic>
#include <memory>
#include <iostream>
#include <thread>
#include <vector>
#include <mutex>
#include <condition_variable>

template<typename T>
class lock_free_queue
{
private:
    struct node;

    struct counted_node_ptr
    {
        int external_count;
        node* ptr;
    };

    std::atomic<counted_node_ptr> head;
    std::atomic<counted_node_ptr> tail;

    struct node_counter
    {
        unsigned internal_count:30;
        unsigned    :2;
    };

    struct node
    {
        std::atomic<T*> data;
        std::atomic<node_counter> count;
        std::atomic<counted_node_ptr> next;

        node()
        {
            node_counter new_count;
            new_count.internal_count = 0;
            new_count.external_counters = 2;
            count.store(new_count);

            data.store(nullptr);
            next.store({0, nullptr});
        }

        void release_ref()
        {
            node_counter old_counter = count.load(std::memory_order_relaxed);
            node_counter new_counter;
            do
            {
                new_counter = old_counter;
                --new_counter.internal_count;
            } while (!count.compare_exchange_strong(old_counter, new_counter,
                                                    std::memory_order_acquire, std::memory_order_relaxed));

            if (new_counter.internal_count == 0 && new_counter.external_counters == 0)
            {
                delete this;
            }
        }
    };

    static void increase_external_count(std::atomic<counted_node_ptr>& counter, counted_node_ptr& old_counter)
    {
        counted_node_ptr new_counter;
        do
        {
            new_counter = old_counter;
            ++new_counter.external_count;
        } while (!counter.compare_exchange_strong(old_counter, new_counter,
                                                  std::memory_order_acquire, std::memory_order_relaxed));

        old_counter.external_count = new_counter.external_count;
    }

    static void free_external_counter(counted_node_ptr &old_node_ptr)
    {
        node* const ptr = old_node_ptr.ptr;
        int const count_increase = old_node_ptr.external_count - 2;

        node_counter old_counter = ptr->count.load(std::memory_order_relaxed);
        node_counter new_counter;
        do
        {
            new_counter = old_counter;
            --new_counter.external_counters;
            new_counter.internal_count += count_increase;
        } while (!ptr->count.compare_exchange_strong(old_counter, new_counter,
                                                     std::memory_order_acquire, std::memory_order_relaxed));

        if (new_counter.internal_count == 0 && new_counter.external_counters == 0)
        {
            delete ptr;
        }
    }

    void set_new_tail(counted_node_ptr &old_tail, counted_node_ptr const &new_tail)
    {
        node* const current_tail_ptr = old_tail.ptr;
        while (!tail.compare_exchange_weak(old_tail, new_tail) && old_tail.ptr == current_tail_ptr);

        if (old_tail.ptr == current_tail_ptr)
            free_external_counter(old_tail);
        else
            current_tail_ptr->release_ref();
    }

public:
    lock_free_queue()
    {
        head.store({0, new node});
        tail.store(head.load());
    }

    ~lock_free_queue()
    {
        while (node* const old_head = head.load().ptr)
        {
            head.store({0, old_head->next.load().ptr});
            delete old_head;
        }
    }

    std::size_t size() const
    {
        int count = 0;
        node* old_head = head.load().ptr;
        while (old_head)
        {
            old_head = old_head->next.load().ptr;
            count++;
        }
        return count;
    }

    std::unique_ptr<T> pop()
    {
        counted_node_ptr old_head = head.load(std::memory_order_relaxed);
        for (;;)
        {
            // 增加头部指针的引用计数
            increase_external_count(head, old_head);

            // 获得指向头部元素的指针
            node* const ptr = old_head.ptr;
            if (ptr == tail.load().ptr)
            {
                return std::unique_ptr<T>();
            }

            // 获得下一个的指针
            counted_node_ptr next = ptr->next.load();

            // 
            if (head.compare_exchange_strong(old_head, next))
            {
                // 这里将头部节点抛出，指针置为nullptr
                T* const res = ptr->data.exchange(nullptr);
                // 减少外部引用
                free_external_counter(old_head);
                return std::unique_ptr<T>(res);
            }

            ptr->release_ref();
        }
    }

    void push(T new_value)
    {
        std::unique_ptr<T> new_data(new T(new_value));
        counted_node_ptr new_next;
        new_next.ptr = new node;
        new_next.external_count = 1;
        counted_node_ptr old_tail = tail.load();

        for (;;)
        {
            //增加尾部指针的引用计数
            increase_external_count(tail, old_tail);
            T* old_data = nullptr;
            if (old_tail.ptr->data.compare_exchange_strong(old_data, new_data.get()))
            {
                counted_node_ptr old_next = {0};
                if (!old_tail.ptr->next.compare_exchange_strong(old_next, new_next))
                {
                    delete new_next.ptr;
                    new_next = old_next;
                }
                set_new_tail(old_tail, new_next);
                new_data.release();
                break;
            }
            else
            {
                counted_node_ptr old_next = {0};
                if (old_tail.ptr->next.compare_exchange_strong(old_next, new_next))
                {
                    old_next = new_next;
                    new_next.ptr = new node;
                }
                set_new_tail(old_tail, old_next);
            }
        }
    }
};

节点的生命周期管理：
多个消费者调用pop操作后，只有一个能够成功获得队列中的元素，如何元素包含指针类型，pop成功的线程得到元素后并将指针释放就很容易发生悬空引用的问题，其他的线程就无法通过队列的next进行迭代获得正确的节点，判空再执行pop操作不符合无锁编程的要求。

1、如何atomic模板类支持为shared_ptr提供无锁实现，那么对生命周期的管理就不是问题了，换言之，直接使用atomic<shared_ptr<>>来实现无锁列队就不会有生命周期的管理难题，前提的得接受使用mutex来提供其具体的实现，那无锁编程也失去了意义。

2、使用延迟队列的方式进行节点生命周期和管理
当线程pop出一个节点时，并不是立即删除释放掉，而是设置一个延迟节点队列，暂时保留节点。通过设置一个原子变量记录访问队列pop的线程数，当存在多个线程并发的时候，可以认为不是节点释放的好时机，当引用计数确定当前线程是唯一一个访问线程时，将延迟队列中的所有节点进行删除。上述的实现也有一定的问题，如果再高并发的问题，延迟队列删除的静态点可能并不容易产生，一定程度上会发生内存泄漏。同时延迟队列的释放可能需要消耗大量的时间，也不可控。在释放时，并发处理pop操作又需要进行同步，添加两个延迟队列进行交换是一个优化的方法，同时增大编码的复杂度。

3、使用引用计数进行节点生命周期的管理

struct counted_node_ptr
{
    int external_count;
    node* ptr;
};

这里使用ptr指向具体的数据节点，external_count为外部引用计数。无锁队列的使用了内外两个引用计数器，在使用pop操作访问一个数据节点的时候，会首先累加外部引用计数，确保在访问期间该节点不会被删除。

static void increase_external_count(std::atomic<counted_node_ptr>& counter, counted_node_ptr& old_counter)
{
    counted_node_ptr new_counter;
    do
    {
        new_counter = old_counter;
        ++new_counter.external_count;
    } while (!counter.compare_exchange_strong(old_counter, new_counter,
                                                std::memory_order_acquire, std::memory_order_relaxed));

    old_counter.external_count = new_counter.external_count;
}

使用counter.compare_exchange_strong(old_counter, new_counter, std::memory_order_acquire, std::memory_order_relaxed)确保累加到当前的头尾指针中。

std::unique_ptr<T> pop()
{
    counted_node_ptr old_head = head.load(std::memory_order_relaxed);
    for (;;)
    {
        // 增加头部指针的引用计数
        increase_external_count(head, old_head);

        // 获得指向头部元素的指针
        node* const ptr = old_head.ptr;
        if (ptr == tail.load().ptr)
        {
            return std::unique_ptr<T>();
        }

        // 获得下一个的指针
        counted_node_ptr next = ptr->next.load();

        // 更新
        if (head.compare_exchange_strong(old_head, next))
        {
            // 这里将头部节点抛出，指针置为nullptr
            T* const res = ptr->data.exchange(nullptr);
            // 减少外部引用
            free_external_counter(old_head);
            return std::unique_ptr<T>(res);
        }

        ptr->release_ref();
    }
}

不断判断获得的旧值与head原子加载的值是否一致判断是狗获得了头部指针，这种情况下，在未修改head的情况下，可能有多个线程访问到了同一个head节点，使用if (head.compare_exchange_strong(old_head, next))进行节点值得更新，获得数据后置空，并减少外部引用计数。更新失败的线程则调用ptr->release_ref()更新内部引用计数。

1、ptr->release_ref()：
立即减少内部引用计数：在 compare_exchange_strong 失败时调用，防止因引用计数不正确导致的内存泄漏或悬空指针问题。
避免死锁：确保立即减少内部引用计数，避免其他线程在尝试释放该节点时等待。

2、free_external_counter(old_head)：
成功更新后减少外部引用计数：在 compare_exchange_strong 成功后调用，确保在所有引用都释放后，可以安全地删除或释放该节点。
确保一致性：外部引用计数的减少必须在成功更新 head 后进行，因为只有在成功更新后，当前线程才真正“持有”对 old_head 的引用。

static void free_external_counter(counted_node_ptr &old_node_ptr)
{
    node* const ptr = old_node_ptr.ptr;
    int const count_increase = old_node_ptr.external_count - 2;

    node_counter old_counter = ptr->count.load(std::memory_order_relaxed);
    node_counter new_counter;
    do
    {
        new_counter = old_counter;
        --new_counter.external_counters;
        new_counter.internal_count += count_increase;
    } while (!ptr->count.compare_exchange_strong(old_counter, new_counter,
                                                    std::memory_order_acquire, std::memory_order_relaxed));

    if (new_counter.internal_count == 0 && new_counter.external_counters == 0)
    {
        delete ptr;
    }
}

eg.个人的一些理解，按照pop的流程，存在多个线程获得head的指针，首先更新头部的线程获得对节点的真正应用，所以调用free_external_counter减少本身外部的引用，并将外部引用原子fetch_add到内部引用中，后续更新失败的线程调用release_ref()更新内部的引用计数，并在外部引用计数和内部引用计数都为0的时候删除节点。

void push(T new_value)
{
    std::unique_ptr<T> new_data(new T(new_value));
    counted_node_ptr new_next;
    new_next.ptr = new node;
    new_next.external_count = 1;
    counted_node_ptr old_tail = tail.load();

    for (;;)
    {
        //增加尾部指针的引用计数
        increase_external_count(tail, old_tail);
        T* old_data = nullptr;
        if (old_tail.ptr->data.compare_exchange_strong(old_data, new_data.get()))
        {
            counted_node_ptr old_next = {0};
            if (!old_tail.ptr->next.compare_exchange_strong(old_next, new_next))
            {
                delete new_next.ptr;
                new_next = old_next;
            }
            set_new_tail(old_tail, new_next);
            new_data.release();
            break;
        }
        else
        {
            counted_node_ptr old_next = {0};
            if (old_tail.ptr->next.compare_exchange_strong(old_next, new_next))
            {
                old_next = new_next;
                new_next.ptr = new node;
            }
            set_new_tail(old_tail, old_next);
        }

首先创建好新的数据节点new_next，使用increase_external_count(tail, old_tail)获得当前的尾部节点，如果尾部节点的数据指针为nullptr，代表当前执行的线程获得了尾部节点，需要进行更新，这里也使用了原子操作old_tail.ptr->data.compare_exchange_strong(old_data, new_data.get())，使用strong类型表示无法接受指令失败的场景。后续判断是否有线程执行了next指针的更新，如果为old_tail.ptr->next.compare_exchange_strong(old_next, new_next)如果为old_next也就是没有其他线程设置，那么将设置好的new_next作为旧old_tail的next指针，并更新tail，否则就释放先前创建的new_next，使用原有的old_tail.ptr->next不作更新操作。同时没有获得tail节点的线程值预先添加tail节点，避免后续的节点分配。
